Usando o ggplot2

Variveis qualitativas

  • Website
  • Toda visualizao precisa de
    • dados,
    • um mapeamento esttico (aes) (varivel -> propriedade visual) e
    • forma de apresentao
# Cada `+`  um layer

g1 = ggplot(dados_2007, aes(x=continent, fill=continent)) + # Mapeamento: continent -> x
  geom_bar() + # Qual o tipo de grfico
  labs( # Renomear legendas
    title = "Pases por continente", 
    x = "", 
    y = "contagem") +
  scale_fill_brewer(palette = "Blues") + # Cores
  theme_bw() + # Tema do grfico
  guides(fill = "none") # Legendas

g1

Variveis quantitativas

g2 = dados_2007 %>%
  ggplot(aes(y=lifeExp, x=continent, fill=continent)) + 
  geom_boxplot()

g2

g3 = dados_2007 %>%
  ggplot(aes(y=gdpPercap, x=lifeExp, color=continent)) + 
  geom_point(
      size=4,
      alpha=0.6) +
  labs(
    title = "Expectativa de vida por PIV per capta", 
    subtitle = "Dados de 2007",
    y = "PIB per capta",
    x ="Expectatviva de vida (anos)") +
  theme_classic()

g3

dados_paises = 
  dados %>%
  filter(country == "Brazil" | country == "Chile" | country == "Paraguay" | country == "Argentina")
  
dados_paises$Pas = recode(dados_paises$country, `Brazil` = "Brasil", `Paraguay` = "Paraguai")

g4 = dados_paises %>%
  ggplot(aes(y=lifeExp, x=year, color=Pas)) +
  geom_line(linewidth=2) +
  labs(
    title = "Expectativa de vida pelo tempo por pas (Argentina, Brasil, Chile, Paraguai)",
    subtitle = "Dados de 1952 a 2007",
    y = "Expectativa de vida",
    x = "Ano"
  ) +
  guides(
    title = "Pases"
  ) +
  theme_bw()
